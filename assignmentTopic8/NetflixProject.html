# Missing data

def checkMissing(df):
    print("null value: ")
    for col in df.columns:
        null = round(df[col].isna().sum() / len(df) * 100, 2)
        if null > 0:
            print(f"{col} null rate: {null}%")


print("sebelum dibersihkan\n")
checkMissing(df)
print(f"shape : {df.shape}")

# drop duplicate

print("\n \nsetelah dibersihkan\n")

print(f"shape : {df.shape}")

df['year'] = df['date_added'].apply(lambda x: x.split(', ')[-1])
df['month'] = df['date_added'].apply(lambda x: x.lstrip().split(' ')[0])

month_order = ['January', 'February', 'March', 'April', 'May', 'June',
               'July', 'August', 'September', 'October', 'November', 'December'][::-1]

plt.figure(figsize=(10, 7))
data = df.groupby('year')['month'].value_counts(
).unstack().fillna(0)[month_order].T.iloc[:, :-1]

plt.pcolor(data, cmap=cmapNetflix, edgecolors='white', linewidths=2)

plt.xticks(np.arange(0.5, len(data.columns), 1), data.columns, fontsize=10)
plt.yticks(np.arange(0.5, len(data.index), 1), data.index, fontsize=10)

          fontsize=16,
          fontweight='bold')

# make input
#@title{ run: "auto" }
year="all time" #@param ['all time',2008, 2009, 2010, 2011, 2012, 2013,2014,2015,2016,2017,2018,2019,2020,2021]

if year == "all time":
  data=df.loc[:,:]
else:
  data=df.loc[df.year==year,:]

# create dictionary
combinedString=""

for e in data['listed_in']:
  combinedString+=e+', '

word=combinedString.split(', ')
word=dict(Counter(word))

# plot wordcloud
plt.figure(figsize=(12, 7))
wordcloud = WordCloud(background_color = 'white', width = 500,  height = 200,colormap=cmapNetflix,relative_scaling=0,max_words=15).generate_from_frequencies(word)
plt.imshow(wordcloud, interpolation="bilinear")
plt.title('Netflix Genre Wordcloud in '+year+'\n',
          fontsize=16,
          fontweight='bold')
plt.axis('off')
plt.show()
